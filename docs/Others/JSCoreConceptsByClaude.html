1. Spread Operator & Shallow Copy

The spread operator {...obj} creates a shallow copy — it only copies the top level of an object. Primitive values (strings, numbers, booleans) get their own independent copies, but nested objects and arrays still share the same memory reference.

  Flat object spread
// Flat object — primitive values are independent copies
const a = { x: 1, y: 2 };
const b = { ...a };

b.x = 99;

console.log(a.x);  // 1    (original unchanged — number copied by value)
console.log(b.x);  // 99
  Nested object — shared array reference
// Nested object — the array is SHARED (shallow copy only goes one level deep)
const user = { name: "Alice", scores: [10, 20, 30] };
const copy = { ...user };

copy.name = "Bob";         // string: copied by value — independent
copy.scores.push(40);      // array: still the same reference!

console.log(user.name);    // "Alice"              (not affected)
console.log(copy.name);    // "Bob"
console.log(user.scores);  // [10, 20, 30, 40]     (changed — shared array!)
console.log(copy.scores);  // [10, 20, 30, 40]     (same array object)
  Deep copy with structuredClone()
// Deep copy with structuredClone() — all levels are truly independent
const user = { name: "Alice", scores: [10, 20, 30] };
const deep = structuredClone(user);

deep.scores.push(99);

console.log(user.scores);  // [10, 20, 30]        (untouched)
console.log(deep.scores);  // [10, 20, 30, 99]
  Merging objects
// Merging objects — properties from later spreads override earlier ones
const defaults = { color: "red", size: "M", weight: 1 };
const custom   = { color: "blue", weight: 2 };
const result   = { ...defaults, ...custom };

console.log(result);  // { color: "blue", size: "M", weight: 2 }
  Array spread
// Spread with arrays
const arr1 = [1, 2, 3];
const arr2 = [4, 5];

const merged = [...arr1, ...arr2];
console.log(merged);  // [1, 2, 3, 4, 5]

const copy = [...arr1];     // shallow copy of array
copy.push(99);
console.log(arr1);          // [1, 2, 3]          (original unchanged)
console.log(copy);          // [1, 2, 3, 99]

2. Destructuring

Destructuring lets you unpack values from objects or arrays directly into named variables. It supports renaming, default values for missing properties, nested patterns, and the rest operator ....

  Basic object destructuring
// Basic object destructuring
const person = { name: "Alice", age: 25, city: "Prague" };
const { name, age } = person;

console.log(name);  // "Alice"
console.log(age);   // 25
// 'city' was not destructured — the variable does not exist here
  Rename  +  default value
// Rename a variable  +  default value for a missing property
const user = { name: "Alice", age: 25, role: "developer" };
const { name: firstName, country: origin = "CZ" } = user;

console.log(firstName);  // "Alice"    (renamed from 'name')
console.log(origin);     // "CZ"       (default — 'country' is not in user)
  Nested destructuring
// Nested destructuring
// IMPORTANT: 'db' is only an alias used to go deeper — it is NOT a variable in scope
const config = {
    db: { host: "localhost", port: 5432 },
    debug: true
};
const { db: { host, port }, debug } = config;

console.log(host);   // "localhost"
console.log(port);   // 5432
console.log(debug);  // true
// console.log(db);  // ReferenceError: db is not defined
  Rest in object destructuring
// Rest operator — collect the remaining properties into a new object
const { a, b, ...rest } = { a: 1, b: 2, c: 3, d: 4 };

console.log(a);     // 1
console.log(b);     // 2
console.log(rest);  // { c: 3, d: 4 }
  Array destructuring
// Array destructuring + rest
const [first, second, ...remaining] = [10, 20, 30, 40, 50];

console.log(first);      // 10
console.log(second);     // 20
console.log(remaining);  // [30, 40, 50]

// Skip elements using empty slots (commas without variable names)
const [, , third] = [10, 20, 30];
console.log(third);      // 30
  Destructuring in function parameters
// Destructuring directly in function parameters
function display({ name, role = "user" }) {
    console.log(`${name} is a ${role}`);
}

display({ name: "Alice", role: "admin" });  // "Alice is a admin"
display({ name: "Bob" });                   // "Bob is a user"   (default applied)
  Variable swap
// Swapping two variables — no temporary variable needed
let x = 1, y = 2;
[x, y] = [y, x];

console.log(x);  // 2
console.log(y);  // 1

3. Promises & Event Loop

JavaScript is single-threaded but non-blocking. The event loop processes tasks in a strict order: synchronous code runs first, then microtasks (Promise .then/.catch callbacks), then macrotasks (setTimeout, setInterval, I/O callbacks).

Queue type

Examples

Runs…

Synchronous code

Regular statements, function calls

1st — before anything async

Microtask queue

Promise .then(), .catch(), .finally()

2nd — after ALL sync code

Macrotask queue

setTimeout(), setInterval(), I/O, events

3rd — after ALL microtasks

  Basic execution order: sync → microtask
// Execution order: sync first, then .then() microtask, then setTimeout macrotask
console.log("start");             // "start"       ①  synchronous

new Promise((resolve) => {
    resolve(42);
}).then((value) => {
    console.log("then:", value);  // "then: 42"    ③  microtask
});

console.log("end");               // "end"         ②  synchronous
  setTimeout(0) vs Promise.resolve()
// setTimeout(fn, 0) is a macrotask — Promise.resolve() microtask still wins
console.log("A");                                  // "A"       ①  sync

setTimeout(() => console.log("B"), 0);             // "B"       ④  macrotask

Promise.resolve().then(() => console.log("C"));    // "C"       ③  microtask

console.log("D");                                  // "D"       ②  sync

// Output order: A  D  C  B
  Promise chaining
// Promise chaining — each .then() receives the return value of the previous one
Promise.resolve(1)
    .then(v => v + 1)             // receives 1, returns 2
    .then(v => v * 2)             // receives 2, returns 4
    .then(v => console.log(v));   // 4
  async / await execution order
// async/await — await pauses only the current async function, NOT the caller
async function run() {
    console.log("inside: before await");    // "inside: before await"   ②
    const val = await Promise.resolve(99);
    console.log("inside: after await");     // "inside: after await"    ④
    return val;
}

console.log("before call");                 // "before call"             ①
run();
console.log("after call");                 // "after call"             ③

// Output order: ① "before call"  ② "inside: before await"  ③ "after call"  ④ "inside: after await"
  Promise.all
// Promise.all — resolves when ALL promises resolve; results preserve input order
const p1 = Promise.resolve(10);
const p2 = Promise.resolve(20);
const p3 = Promise.resolve(30);

Promise.all([p1, p2, p3]).then(values => {
    console.log(values);  // [10, 20, 30]
});

// If ANY single promise rejects, Promise.all rejects immediately
Promise.all([Promise.resolve(1), Promise.reject("oops")])
    .catch(err => console.log("rejected:", err));  // "rejected: oops"
  .catch() and .finally()
// Error handling: .catch() recovers the chain, .finally() always runs
Promise.reject(new Error("something went wrong"))
    .catch(err => {
        console.log("caught:", err.message);  // "caught: something went wrong"
        return "recovered";                   // chain continues with this value
    })
    .then(val => console.log(val))            // "recovered"
    .finally(() => console.log("done"));     // "done"  (runs on both success and error)

4. this — Regular vs Arrow Functions

this in a regular function is determined at call time — it is set to whatever object is before the dot when the function is called. Arrow functions have no own this; they capture this from the surrounding lexical scope at the moment they are defined.

  Object method: regular vs arrow
// Regular method: this = the calling object  |  Arrow: this = outer scope
const obj = {
    name: "Alice",
    greetRegular: function() {
        return "Hello, " + this.name;    // this = obj (set at call time)
    },
    greetArrow: () => {
        return "Hello, " + this.name;   // this = module/global scope
    }
};

console.log(obj.greetRegular());  // "Hello, Alice"
console.log(obj.greetArrow());    // "Hello, undefined"
  Extracted method loses this
// this is LOST when a regular method is extracted from its object
const obj = {
    name: "Alice",
    greet() { return "Hi, " + this.name; }
};

const fn = obj.greet;       // extracted reference — no longer tied to obj
console.log(fn());          // "Hi, undefined"    (this = global/undefined)
console.log(obj.greet());   // "Hi, Alice"         (called on obj: this = obj)
  Arrow in setTimeout — fixing the this problem
// Arrow in setTimeout captures this from the enclosing method — no binding lost
const counter = {
    count: 0,

    startBroken: function() {
        setTimeout(function() {
            this.count++;               // this = global (strict: undefined)
            console.log(this.count);    // NaN  /  TypeError in strict mode
        }, 0);
    },

    startFixed: function() {
        setTimeout(() => {
            this.count++;               // arrow inherits this from startFixed
            console.log(this.count);    // 1
        }, 0);
    }
};

counter.startFixed();  // 1
  call() / apply() / bind()
// Explicit this binding with call(), apply(), bind()
function greet(greeting, punct) {
    return `${greeting}, ${this.name}${punct}`;
}
const user = { name: "Alice" };

// call:  pass this + arguments individually
console.log(greet.call(user, "Hello", "!"));    // "Hello, Alice!"

// apply: pass this + arguments as an array
console.log(greet.apply(user, ["Hi", "?"]));    // "Hi, Alice?"

// bind:  returns a NEW function with this permanently fixed
const sayHey = greet.bind(user, "Hey");
console.log(sayHey("..."));                     // "Hey, Alice..."
  Class field arrow vs prototype method
// Class field arrow vs prototype method
class Timer {
    ticks = 0;

    // Arrow class field: 'this' is always the instance, even when extracted
    tick = () => {
        this.ticks++;
    };

    // Prototype method: 'this' depends on call site
    reset() {
        this.ticks = 0;
    }
}

const t = new Timer();

const fn = t.tick;     // extract the arrow field
fn();                  // safe — arrow keeps this = t
console.log(t.ticks);  // 1

const reset = t.reset; // extract the prototype method
// reset();            // TypeError in strict mode — this is undefined
